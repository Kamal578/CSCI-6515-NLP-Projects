\section{Motivation}

\begin{frame}{Motivation and Objective}
\begin{itemize}
  \item LLM-only answers are fluent but often not grounded in project data.
  \item ESG analysis needs traceability: answers should cite real source reports.
  \item Goal: build a full RAG system from data prep to UI.
  \item Required flow:
  \[
  \text{Query} \rightarrow \text{Retrieval} \rightarrow \text{Context} \rightarrow \text{Answer}
  \]
\end{itemize}
\end{frame}

\section{Task 1: Data Preparation}

\begin{frame}{Domain and Dataset}
\small
\textbf{Domain}
\begin{itemize}
  \item ESG sustainability reports for S\&P 500 companies
\end{itemize}

\textbf{Dataset source}
\begin{itemize}
  \item Kaggle: ESG Sustainability Reports of S\&P 500 Companies
\end{itemize}

\textbf{Local snapshot}
\begin{itemize}
  \item 866 documents
  \item 10,361,942 words
  \item requirement exceeded (\(>50\) docs and \(>10,000\) words)
\end{itemize}
\end{frame}

\begin{frame}{Preprocessing and Chunking}
\small
\textbf{Pipeline}
\begin{itemize}
  \item \texttt{01\_data\_prep.py}: CSV rows \(\rightarrow\) per-report TXT files
  \item \texttt{02\_build\_index.py}: tokenization, strict chunking, indexing
\end{itemize}

\textbf{Chunk policy}
\begin{itemize}
  \item tokenization: whitespace regex \texttt{\textbackslash S+}
  \item strict token bounds: 201 to 499
  \item target chunk size: 420
  \item adaptive chunk count:
  \[
  k=\min\!\Big(\max(\text{round}(n/420),\lceil n/499\rceil),\lfloor n/201\rfloor\Big)
  \]
  \item stored chunks: 24,687
  \item observed min/max token count: 215 / 488
\end{itemize}
\end{frame}

\section{Task 2: RAG Implementation}

\begin{frame}{System Architecture}
\small
\textbf{Knowledge Base}
\begin{itemize}
  \item ChromaDB persistent collection: \texttt{sp500\_esg\_reports}
\end{itemize}

\textbf{Retriever}
\begin{itemize}
  \item LlamaIndex similarity search, default \texttt{top-k = 3}
\end{itemize}

\textbf{Generator}
\begin{itemize}
  \item Ollama \texttt{llama3.1} with retrieval-aware prompts
  \item shared prompt module: \texttt{src/rag\_prompts.py}
\end{itemize}

\textbf{Embeddings}
\begin{itemize}
  \item Ollama \texttt{nomic-embed-text}
\end{itemize}
\end{frame}

\begin{frame}{System Prompt Contract}
\small
\textbf{Prompt source}
\begin{itemize}
  \item \texttt{ESG\_SYSTEM\_PROMPT} in \texttt{src/rag\_prompts.py}
  \item applied in both pipeline and UI
\end{itemize}

\textbf{Core rules}
\begin{itemize}
  \item use retrieved chunks as the only authoritative evidence
  \item do not fabricate facts/citations
  \item separate evidence from inference
  \item state uncertainty when evidence is weak
\end{itemize}

\textbf{Required output format}
\begin{itemize}
  \item \texttt{Answer}
  \item \texttt{Evidence}
  \item optional \texttt{Inference}
  \item optional \texttt{Gaps/Uncertainty}
  \item citation format: \texttt{[source\_file | ticker | year]}
\end{itemize}
\end{frame}

\begin{frame}{Embeddings: Ollama \texttt{nomic-embed-text}}
\small
\textbf{Embedding model}
\begin{itemize}
  \item Ollama embedding model: \texttt{nomic-embed-text}
  \item encoder map: \(f_{\theta}: \mathcal{T} \rightarrow \mathbb{R}^{d}\), \(e_i=f_{\theta}(c_i)\)
  \item optional normalization: \(\hat{e}_i = e_i / \|e_i\|_2\)
\end{itemize}

\textbf{Where used}
\begin{itemize}
  \item indexing: \texttt{src/02\_build\_index.py}
  \item retrieval pipeline: \texttt{src/03\_rag\_pipeline.py}
  \item UI runtime: \texttt{src/04\_app.py}
\end{itemize}

\textbf{Stored with each vector}
\begin{itemize}
  \item chunk text embedding + metadata
  \item \texttt{source\_file}, \texttt{ticker}, \texttt{year}, \texttt{chunk\_id}, \texttt{token\_count}
\end{itemize}
\end{frame}

\begin{frame}{Baseline vs RAG Evaluation Setup}
\small
\textbf{Baseline path}
\begin{itemize}
  \item direct LLM answer without retrieval
\end{itemize}

\textbf{RAG path}
\begin{itemize}
  \item retrieve top-k chunks \(\rightarrow\) generate grounded answer
  \item score chunks by cosine similarity:
  \[
  s_i=\cos(e_q,e_i)=\frac{e_q^\top e_i}{\|e_q\|_2\|e_i\|_2}
  \]
  \item retrieve \(\mathcal{I}_k(q)=\operatorname{TopK}_{i}(s_i)\), with \(k=3\)
\end{itemize}

\textbf{Run used in this presentation (2026-04-23)}
\begin{itemize}
  \item 15 precomputed evaluation questions
  \item file: \texttt{rag\_eval\_20260423T113135Z.json}
  \item same model family for fair comparison
\end{itemize}
\end{frame}

\section{Discussion}

\begin{frame}{Error Analysis and Limitations}
\small
\textbf{Common issues}
\begin{itemize}
  \item some chunks include noisy text from report preprocessing
  \item occasional over-inference beyond explicit evidence
\end{itemize}

\textbf{Current limitations}
\begin{itemize}
  \item metrics are mostly structural, not full human-judged factuality
  \item data cleaning is partly inherited from upstream CSV processing
\end{itemize}

\textbf{Next steps}
\begin{itemize}
  \item add reranking and citation verification
  \item add manual answer grading protocol
  \item test adaptive retrieval depth
\end{itemize}
\end{frame}
